program MyDef;

{$APPTYPE CONSOLE}

const
  M = 5;
  N = 5;

var
  A: array [1..M,1..N] of real;  //Матрица А
  X: array [1..M] of real;  //Массив для средних арифметических чисел
  row, col: integer;  //Реальное число строк и столбцов
  i, j: integer;  //Параметры для циклов
  maxsr: real;  //Максимальное среднее арифметическое

begin
  repeat
    begin
      write('Введите количество строк матрицы X: '); // Ввод исходных данных
      readln(row);
      writeln;
      write('Введите количество столбцов матрицы X: ');
      readln(col);
      writeln;
      writeln('Введите матрицу A[',row,',',col,']: ');

      for i := 1 to row do
        begin
          for j := 1 to col do
            read(A[i,j]);
          readln;
        end;
    end;
  until (M > 0)and(N > 0)and(M < 6)and(N < 6);

  writeln;
  writeln('Матрица X: '); // Вывод исходных данных
  for i:=1 to str do
    begin
      for j:=1 to col do
        write(A[i,j],'  ');
      writeln;
    end;

  for i:=1 to row do  //Вычисление средних арифметических в строках матрицы А
    begin
      X[i]=0
      for j:=1 to col do
        X[i]=X[i]+A[i,j]
      X[i]=X[i]/col   //Сохраняем значения в массиве X
    end;

  maxsr=X[0]
  for i:=2 to M do  //Вычисление наибольшего среднего арифметического
    if X[i]>maxsr:
      maxsr=X[i]

  writeln; 
  for i:=1 to str do  //Вывод средних арифметических чисел в строках матрицы А
    begin
      writeln('Среднее арифметическое элементов в ',i,' строке: ',X[i,j],'  ');
    end;
  
  writeln;  //Вывод максимального среднего арифметического числа
  writeln('Максимальное среднее арифметическое: ',maxsr,'  ');

end.
